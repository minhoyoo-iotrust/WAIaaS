    Enhancement approach -- add a helper function to derive display label from transaction metadata:

    1. In transactions.tsx, add a helper function (near the existing formatAddress helper):
       ```typescript
       /**
        * Derive a display label for CONTRACT_CALL transactions from action provider metadata.
        * For action-provider-originated transactions, the calldata contains provider info.
        */
       function deriveActionLabel(row: TxRow): string | null {
         if (row.type !== 'CONTRACT_CALL') return null;
         // Check if contractName is already resolved (from ContractNameRegistry)
         if (row.contractName) return null; // let existing display handle it
         // Try to parse calldata for XRPL DEX signature
         try {
           const meta = row.metadata as Record<string, unknown> | undefined;
           if (meta?.actionProvider === 'xrpl_dex') {
             const actionNames: Record<string, string> = {
               swap: 'XRPL DEX Swap',
               limit_order: 'XRPL DEX Limit Order',
               cancel_order: 'XRPL DEX Cancel Order',
             };
             return actionNames[meta.actionName as string] ?? 'XRPL DEX';
           }
         } catch { /* ignore */ }
         return null;
       }
       ```

We can enhance the Admin UI display to show "Contract Call" instead of raw "CONTRACT_CALL" type label, and when contractName is available show it. This is already the current behavior.
