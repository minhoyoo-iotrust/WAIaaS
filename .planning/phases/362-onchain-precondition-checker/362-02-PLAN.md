```typescript
export interface CheckResult {
  name: string;
  passed: boolean;
  required: string;
  actual: string;
  message: string;
}

export interface PreconditionReport {
  allPassed: boolean;
  checks: CheckResult[];
  summary: string;
}

export interface NetworkFilter {
  networks?: string[];
  protocols?: string[];
}

export class PreconditionChecker {
  constructor(baseUrl: string, masterPassword: string);
  checkDaemon(): Promise<CheckResult>;
  checkWallets(requiredChains: string[]): Promise<CheckResult[]>;
  checkBalance(walletId: string, minBalance: string, network: string): Promise<CheckResult>;
  runAll(filter?: NetworkFilter): Promise<PreconditionReport>;
  generateReport(report: PreconditionReport): string;
}
```
