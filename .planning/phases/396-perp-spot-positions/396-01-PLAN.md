async getPositions(walletId: string): Promise<PositionUpdate[]> {

  const now = Math.floor(Date.now() / 1000);
  return positions.map((pos) => ({
    walletId,
    category: 'PERP' as PositionCategory,
    provider: 'drift_perp',
    chain: 'solana',
    network: 'solana-mainnet',
    assetId: null,
    amount: pos.baseAssetAmount,
    amountUsd: pos.notionalValueUsd,
    metadata: { direction, leverage, unrealizedPnl, liquidationPrice, margin, entryPrice, market },
    status: 'ACTIVE' as const,
    openedAt: now,

}
