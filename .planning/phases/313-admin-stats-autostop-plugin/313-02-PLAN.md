         counters: Record<string, number>;
         latencies: Record<string, { count: number; totalMs: number; avgMs: number }>;
       }

       export interface IMetricsCounter {
         increment(key: string, labels?: Record<string, string>): void;
         recordLatency(key: string, durationMs: number, labels?: Record<string, string>): void;
         getCount(key: string, labels?: Record<string, string>): number;
         getAvgLatency(key: string, labels?: Record<string, string>): number;
         snapshot(): MetricsSnapshot;
         reset(): void;
       }

    2. Update `packages/core/src/index.ts` to export IMetricsCounter and MetricsSnapshot.

    3. Create `packages/daemon/src/infrastructure/metrics/in-memory-counter.ts`:
       - Map<string, number> for counters (key includes serialized labels: `rpc.calls|network=solana-mainnet`)
       - Map<string, { count: number; totalMs: number }> for latencies
       - Private `makeKey(key, labels?)`: joins labels as sorted key=value pairs separated by `|`
       - `increment(key, labels?)`: counters.set(k, (counters.get(k) || 0) + 1)
       - `recordLatency(key, durationMs, labels?)`: accumulate count+totalMs
       - `getCount(key, labels?)`: return counters.get(k) || 0
       - `getAvgLatency(key, labels?)`: return totalMs/count or 0
       - `snapshot()`: build MetricsSnapshot from all entries
       - `reset()`: clear both maps
       - `getCountsByPrefix(prefix)`: return entries whose key starts with prefix (used by AdminStatsService to get per-network RPC stats)

    4. Create `packages/daemon/src/services/admin-stats-service.ts`:
       ```typescript
       export class AdminStatsService {
         private cache: AdminStatsResponse | null = null;
         private cacheExpiry = 0;
         private readonly TTL_MS = 60_000; // 1 minute

         constructor(private opts: {
           sqlite: SQLiteDatabase;
           metricsCounter: IMetricsCounter;
           autoStopService?: { getStatus(): any };
           startTime: number;
           version: string;
           dataDir?: string;
         }) {}

         getStats(): AdminStatsResponse { ... }
         invalidateCache(): void { this.cache = null; }
       }
       ```
       - `getStats()`: if cache valid, return cache. Otherwise:
         a. **transactions**: raw SQL `SELECT status, COUNT(*) FROM transactions GROUP BY status`, `SELECT type, COUNT(*) FROM transactions GROUP BY type`, plus 24h/7d window queries
         b. **sessions**: `SELECT COUNT(*) FROM sessions WHERE revoked_at IS NULL AND expires_at > ?` for active, total count, revoked last 24h
         c. **wallets**: `SELECT status, COUNT(*) FROM wallets GROUP BY status`, `SELECT COUNT(*) FROM wallets WHERE owner_address IS NOT NULL`
         d. **rpc**: from metricsCounter -- getCount('rpc.calls'), getCount('rpc.errors'), getAvgLatency('rpc.latency'), plus byNetwork from getCountsByPrefix
         e. **autostop**: from autoStopService.getStatus() -- enabled, triggeredTotal from audit_log COUNT WHERE event_type='AUTO_STOP_TRIGGERED', rules from registry status, lastTriggeredAt from audit_log MAX
         f. **notifications**: `SELECT status, COUNT(*) FROM notification_logs WHERE created_at >= ? GROUP BY status` for 24h, channelStatus from settings
         g. **system**: process.uptime() * 1000, version, LATEST_SCHEMA_VERSION, fs.statSync(db path).size, process.version, process.platform, Date.now()/1000
       - Use `this.opts.sqlite.prepare(...).all()` for queries (better-sqlite3 sync, matching existing patterns)
       - Store result in cache with expiry timestamp
