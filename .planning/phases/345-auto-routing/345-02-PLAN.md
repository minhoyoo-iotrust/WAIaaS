    1. Currently, `resolve('dex_swap')` calls `executeDexSwap()` directly which throws on no-route. Add auto-routing fallback:
       ```typescript
       case 'dex_swap': {
         const input = DexSwapInputSchema.parse(params);
         try {
           return await executeDexSwap(this.getClient(), { ...input, walletAddress: context.walletAddress }, this.config);
         } catch (err) {
           // DS-04: Fallback to 2-hop auto-routing when direct route unavailable
           if (err instanceof ChainError && isNoRouteError(err)) {
             const twoHopResult = await executeTwoHopSwap(
               this.getClient(),
               { ...input, walletAddress: context.walletAddress },
               this.config,
             );
             // Attach metadata to requests for pipeline transparency
             return twoHopResult.requests;
           }
           throw err;
         }
       }
       ```
