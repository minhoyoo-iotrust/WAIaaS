```typescript
// UserOperationV07Schema -- EntryPoint v0.7 전체 필드 (SCHM-05)
// 모든 필드는 hex string (0x 접두사). gas 필드는 플랫폼이 채우므로 optional이 아닌 required.
export const UserOperationV07Schema = z.object({
  sender: z.string().regex(/^0x[0-9a-fA-F]{40}$/),
  nonce: z.string().regex(/^0x/),
  callData: z.string().regex(/^0x/),
  callGasLimit: z.string().regex(/^0x/),
  verificationGasLimit: z.string().regex(/^0x/),
  preVerificationGas: z.string().regex(/^0x/),
  maxFeePerGas: z.string().regex(/^0x/),
  maxPriorityFeePerGas: z.string().regex(/^0x/),
  signature: z.string().regex(/^0x/),
  // Optional init fields (for undeployed accounts)
  factory: z.string().regex(/^0x[0-9a-fA-F]{40}$/).optional(),
  factoryData: z.string().regex(/^0x/).optional(),
  // Optional paymaster fields
  paymaster: z.string().regex(/^0x[0-9a-fA-F]{40}$/).optional(),
  paymasterData: z.string().regex(/^0x/).optional(),
  paymasterVerificationGasLimit: z.string().regex(/^0x/).optional(),
  paymasterPostOpGasLimit: z.string().regex(/^0x/).optional(),
});

// UserOpBuildRequestSchema -- TransactionRequest 재사용 + network 필수 (SCHM-01)
export const UserOpBuildRequestSchema = z.object({
  request: TransactionRequestSchema,
  network: NetworkTypeEnum,
});

// UserOpBuildResponseSchema -- unsigned UserOp 골격 (SCHM-02)
// gas/paymaster 필드 미포함 (BUILD-11)
export const UserOpBuildResponseSchema = z.object({
  sender: z.string(),
  nonce: z.string(),
  callData: z.string(),
  factory: z.string().nullable(),
  factoryData: z.string().nullable(),
  entryPoint: z.string(),
  buildId: z.string().uuid(),
});

// UserOpSignRequestSchema -- buildId + 완성된 UserOp (SCHM-03)
export const UserOpSignRequestSchema = z.object({
  buildId: z.string().uuid(),
  userOperation: UserOperationV07Schema,
});

// UserOpSignResponseSchema (SCHM-04)
export const UserOpSignResponseSchema = z.object({
  signedUserOperation: UserOperationV07Schema,
  txId: z.string().uuid(),
});
```
