**CompletionWaiter** (`packages/daemon/src/rpc-proxy/completion-waiter.ts`):

Following Architecture Pattern 1 exactly. Single class:

```
class CompletionWaiter {
  private pending = new Map<string, { resolve, reject, timer }>();

  constructor(eventBus: EventBus) {
    // Register TWO global listeners on construction
    eventBus.on('transaction:completed', (ev) => { ... });
    eventBus.on('transaction:failed', (ev) => { ... });
  }

  waitForCompletion(txId: string, timeoutMs: number): Promise<string> { ... }
  dispose(): void { ... }
}
```

Key: On `transaction:completed` event, look up pending Map by ev.txId, clearTimeout, resolve(ev.txHash), delete entry. On `transaction:failed`, reject with Error(ev.error). On timeout, reject with descriptive message including txId and timeout duration.

```typescript
import { WAIaaSError } from '@waiaas/core';
import {
  stage1Validate, stage2Auth, stage3Policy, stage3_5GasCondition,
  stage4Wait, stage5Execute, stage6Confirm,
  type PipelineContext
} from '../pipeline/stages.js';
import type { CompletionWaiter } from './completion-waiter.js';
import type { SettingsService } from '../infrastructure/settings/settings-service.js';

export class SyncPipelineExecutor {
  constructor(
    private completionWaiter: CompletionWaiter,
    private settingsService?: SettingsService,
  ) {}

  async execute(ctx: PipelineContext): Promise<string> {
    // Set source for audit trail (SEC-04 preparation for Phase 400)
    (ctx as any).source = 'rpc-proxy';

    // Stages 1-3.5: always run directly
    await stage1Validate(ctx);
    await stage2Auth(ctx);
    await stage3Policy(ctx);
    await stage3_5GasCondition(ctx);

    try {
      await stage4Wait(ctx);
    } catch (err) {
      if (err instanceof WAIaaSError && err.code === 'PIPELINE_HALTED') {
        // DELAY/APPROVAL: delegate to CompletionWaiter
        const timeoutMs = this.resolveTimeout(ctx.tier);
        return this.completionWaiter.waitForCompletion(ctx.txId, timeoutMs);
      }
      throw err; // Non-HALTED errors propagate
    }

    // INSTANT: stages 5-6 directly
    await stage5Execute(ctx);
    await stage6Confirm(ctx);

    return ctx.submitResult?.txHash ?? '';
  }

  private resolveTimeout(tier?: string): number {
    if (tier === 'DELAY') {
      const s = this.settingsService?.get('rpc_proxy.delay_timeout_seconds');
      return (s ? parseInt(s, 10) : 300) * 1000;
    }
    const s = this.settingsService?.get('rpc_proxy.approval_timeout_seconds');
    return (s ? parseInt(s, 10) : 600) * 1000;
  }
}
```

**Critical**: Do NOT modify stage4Wait or any existing pipeline code. The PIPELINE_HALTED catch pattern keeps existing fire-and-forget flow untouched (Anti-Pattern 1 in Architecture doc).
