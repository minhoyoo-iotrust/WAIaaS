```typescript
import { serializeTransaction, encodeFunctionData } from 'viem';
// native transfer
const nativeTx = serializeTransaction({ type: 'eip1559', to: '0x...', value: 1000n, chainId: 1, nonce: 0, gas: 21000n, maxFeePerGas: 1n, maxPriorityFeePerGas: 1n });
// token transfer (ERC-20 transfer selector)
const transferData = encodeFunctionData({ abi: [...], functionName: 'transfer', args: [...] });
const tokenTx = serializeTransaction({ type: 'eip1559', to: tokenAddr, data: transferData, ... });
// approve
const approveData = encodeFunctionData({ abi: [...], functionName: 'approve', args: [...] });
const approveTx = serializeTransaction({ type: 'eip1559', to: tokenAddr, data: approveData, ... });
// contract call (arbitrary selector)
const contractTx = serializeTransaction({ type: 'eip1559', to: contractAddr, data: '0xdeadbeef' + '00'.repeat(28), ... });
```
