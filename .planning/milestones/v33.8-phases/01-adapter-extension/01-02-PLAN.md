1. In `tx-parser.ts`, add two new `else if` branches before the final `else` (UNKNOWN) block:

   **OfferCreate branch** (after TrustSet, before else):
   ```typescript
   } else if (txType === 'OfferCreate') {
     // DEX offer: TakerGets is what the Account gives (= spending amount)
     const takerGets = tx['TakerGets'];
     let parsedAmount: bigint | undefined;
     let token: string | undefined;

     if (typeof takerGets === 'string') {
       // XRP in drops
       parsedAmount = BigInt(takerGets);
     } else if (typeof takerGets === 'object' && takerGets !== null) {
       const iou = takerGets as { currency?: string; issuer?: string; value?: string };
       token = `${iou.currency ?? 'unknown'}.${iou.issuer ?? 'unknown'}`;
       try {
         parsedAmount = iouToSmallestUnit(iou.value ?? '0', IOU_DECIMALS);
       } catch {
         parsedAmount = 0n;
       }
     }

     operations.push({
       type: 'CONTRACT_CALL',
       amount: parsedAmount,
       token,
       method: 'OfferCreate',
     });
   ```

   **OfferCancel branch:**
   ```typescript
   } else if (txType === 'OfferCancel') {
     operations.push({
       type: 'CONTRACT_CALL',
       method: 'OfferCancel',
     });
   ```

2. In `tx-parser.test.ts`:

   **Remove** the existing test 'parses unknown TransactionType as UNKNOWN' (lines 62-72) which uses OfferCreate as the "unknown" example. Replace it with a test using a genuinely unknown type like 'AccountDelete'.
