    4. In wallets.ts, create exported helper function buildProviderStatus:
       ```typescript
       export function buildProviderStatus(wallet: { aaProvider: string | null; aaPaymasterUrl?: string | null }): ProviderStatus | null {
         if (!wallet.aaProvider) return null;
         const provider = wallet.aaProvider as AaProviderName;
         if (provider === 'custom') {
           return { name: 'custom', supportedChains: [], paymasterEnabled: !!wallet.aaPaymasterUrl };
         }
         // preset: derive supportedChains from AA_PROVIDER_CHAIN_MAP
         const chains = Object.keys(AA_PROVIDER_CHAIN_MAP[provider]);
         return { name: provider, supportedChains: chains, paymasterEnabled: true };
       }
       ```
