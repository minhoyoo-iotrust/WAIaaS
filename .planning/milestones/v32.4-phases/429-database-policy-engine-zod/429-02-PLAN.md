```typescript
/**
 * Parse policy rules JSON with Zod validation.
 * Throws POLICY_RULES_CORRUPT on invalid JSON or schema mismatch.
 */
private parseRules<T>(rules: string, schema: z.ZodType<T>, policyType: string): T {
  const result = safeJsonParse(rules, schema);
  if (!result.success) {
    throw new WAIaaSError('POLICY_RULES_CORRUPT', `${policyType} policy rules corrupt: ${result.error.message}`);
  }
  return result.data;
}
```
