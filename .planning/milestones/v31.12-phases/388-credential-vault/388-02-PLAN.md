export const CreateCredentialParamsSchema = z.object({

  name: z.string().min(1).max(128),
  value: z.string().min(1),
  metadata: z.record(z.unknown()).optional(),
  expiresAt: z.number().int().positive().optional(),
});
export const CredentialMetadataSchema = z.object({
  id: z.string(), walletId: z.string().nullable(), type: CredentialTypeEnum,
  name: z.string(), metadata: z.record(z.unknown()),

});

  <behavior>
    --- Per-wallet endpoints ---
    - GET /v1/wallets/:walletId/credentials (sessionAuth) -> 200 CredentialMetadata[] (value never included)
    - POST /v1/wallets/:walletId/credentials (masterAuth) -> 201 CredentialMetadata (body: CreateCredentialParams)
    - DELETE /v1/wallets/:walletId/credentials/:ref (masterAuth) -> 204
    - PUT /v1/wallets/:walletId/credentials/:ref/rotate (masterAuth) -> 200 CredentialMetadata (body: { value: string })

    --- Global endpoints ---
    - GET /v1/admin/credentials (masterAuth) -> 200 CredentialMetadata[]
    - POST /v1/admin/credentials (masterAuth) -> 201 CredentialMetadata
    - DELETE /v1/admin/credentials/:ref (masterAuth) -> 204
    - PUT /v1/admin/credentials/:ref/rotate (masterAuth) -> 200 CredentialMetadata

    --- Error cases ---
    - POST with duplicate name -> 409 Conflict
    - DELETE/rotate non-existent -> 404 CREDENTIAL_NOT_FOUND
    - GET list on non-existent walletId -> 200 [] (empty, not 404)
    - POST with invalid type -> 400 validation error
    - sessionAuth on write endpoints -> 401/403
  </behavior>
