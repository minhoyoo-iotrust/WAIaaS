executeSignedDataAction(deps, action: SignedDataAction) -> Promise<{ id: string; status: string }>:
  a. Generate UUID v7 txId via generateId()
  b. credentialRef가 있으면 credentialVault.get(credentialRef, walletId)로 복호화
  c. policyContext가 있으면 TransactionParam 매핑 후 policyEngine.evaluateAction() 호출 (DENIED이면 WAIaaSError('POLICY_DENIED'))
  d. transactions INSERT: id=txId, walletId, type='CONTRACT_CALL', status='PENDING', toAddress='external:'+venue, amount='0', action_kind='signedData', venue, operation, metadata={provider, action, payload 요약}
  e. SigningParams 구성: scheme별로 적절한 variant 생성. credential이 있으면 credential 바이트를 params에 포함.
     - eip712: { scheme: 'eip712', privateKey, domain, types, primaryType, value } (payload에서 추출)
     - hmac-sha256: { scheme: 'hmac', credential (Buffer), message (payload를 canonical JSON) }
     - personal-sign: { scheme: 'personal', privateKey, message }
     - 등 scheme별 분기
  f. keyStore.getPrivateKey(walletId, masterPassword) → privateKey. try/finally로 keyStore.releaseKey()
  g. signerRegistry.resolve(action).sign(signingParams) → SigningResult
  h. transactions UPDATE: status='CONFIRMED', txHash='sig:'+signature첫32바이트hex
  i. tracking 필드가 있으면 bridgeStatus='PENDING', bridgeMetadata={trackerName: tracking.trackerName, ...tracking.metadata, enrolledAt: Date.now()}
  j. insertAuditLog(sqlite, { eventType: 'ACTION_SIGNED', actor: sessionId ?? 'system', walletId, txId, details: { venue, operation, signingScheme }, severity: 'INFO' })
  k. return { id: txId, status: 'CONFIRMED' }

executeSignedHttpAction(deps, action: SignedHttpAction, provider?: IActionProvider) -> Promise<{ id: string; status: string }>:
  유사한 흐름이나 차이점:
  d. action_kind='signedHttp'
  g. 서명 후 signedRequest = { ...action, headers: { ...action.headers, ...signingResult.metadata } }
  h. provider?.execute가 있으면 provider.execute(action.actionName ?? '', signedRequest, actionContext) 호출 → executeResult
     provider.execute 없으면 signedRequest만 반환 (클라이언트가 직접 HTTP 전송)
  i. transactions UPDATE: status, external_id = executeResult?.externalId
  j. insertAuditLog eventType='ACTION_HTTP_SIGNED'
