     ```typescript
       id: string;           // 고유 ID (예: 'auth-session-crud')
       name: string;         // 사람이 읽을 수 있는 이름
       track: Track;         // 'offchain' | 'onchain'
       category: string;     // 분류 (예: 'core', 'interface', 'advanced', 'transfer')
       description: string;  // 시나리오 설명

       networks?: string[];
       /** 온체인 시나리오의 경우 필요한 프로토콜 */
       protocols?: string[];
     }
     ```
   - `ScenarioResult` 인터페이스:
     ```typescript
     export interface ScenarioResult {
       scenario: E2EScenario;
       status: ScenarioStatus;
       durationMs: number;
       error?: string;
       skipReason?: string;
     }
     ```
   - `ScenarioRegistry` 클래스:
     ```typescript

       private scenarios: Map<string, E2EScenario> = new Map();

       get(id: string): E2EScenario | undefined;
       getByTrack(track: Track): E2EScenario[];
       getByCategory(category: string): E2EScenario[];
       all(): E2EScenario[];
     }
     ```
   - 글로벌 registry 인스턴스를 export: `export const registry = new ScenarioRegistry()`
