```typescript
// STS-01: Enroll Across bridge execute transactions in async tracking
// AsyncPollingService.pollAll() picks up bridge_status='PENDING' with tracker='across-bridge'
if (provider === 'across_bridge' && action === 'execute') {
  // Extract bridge metadata from the execute params for status tracking
  const execParams = body.params ?? {};
  const originChainId = (() => {
    try {
      const { getAcrossChainId } = require('@waiaas/actions');
      return getAcrossChainId(String(execParams.fromChain ?? ''));
    } catch { return undefined; }
  })();
  const destChainId = (() => {
    try {
      const { getAcrossChainId } = require('@waiaas/actions');
      return getAcrossChainId(String(execParams.toChain ?? ''));
    } catch { return undefined; }
  })();

  // Get the confirmed TX hash from the pipeline result
  const confirmedTx = await deps.db
    .select()
    .from(transactions)
    .where(eq(transactions.id, ctx.txId))
    .get();
  const txHash = confirmedTx?.txHash ?? null;

  if (txHash) {
    await deps.db
      .update(transactions)
      .set({
        bridgeStatus: 'PENDING',
        bridgeMetadata: JSON.stringify({
          tracker: 'across-bridge',
          txHash,
          originChainId: originChainId ?? null,
          destChainId: destChainId ?? null,
          inputToken: String(execParams.inputToken ?? ''),
          outputToken: String(execParams.outputToken ?? ''),
          inputAmount: String(execParams.amount ?? ''),
          notificationEvent: 'BRIDGE_COMPLETED',
          enrolledAt: Date.now(),
        }),
      })
      .where(eq(transactions.id, ctx.txId));
  }
}
```

**중요**: `require('@waiaas/actions')` 대신 dynamic import 사용. actions route 파일 상단에 이미 `@waiaas/actions`가 import되어 있으면 그것을 사용한다. 아니라면 `getAcrossChainId`를 사용하지 않고 `body.params`에서 fromChain/toChain 문자열을 그대로 metadata에 저장하는 것도 허용한다 (tracker의 checkStatus는 originChainId를 optional로 처리하므로).

실제 구현 시:
- `getAcrossChainId` import 대신, bridge enrollment에서는 chainId 변환을 skip하고 metadata에 chain name을 저장해도 된다. AcrossBridgeStatusTracker.checkStatus()에서 metadata.originChainId가 number면 사용하고, 없으면 depositTxnRef(txHash)만으로 조회한다 (Across API는 originChainId 없이도 동작).
- 더 간결한 구현: `originChainId`/`destChainId` 변환을 enrollment에서 하지 않고, fromChain/toChain 문자열을 그대로 저장. tracker에서 필요 시 변환.
