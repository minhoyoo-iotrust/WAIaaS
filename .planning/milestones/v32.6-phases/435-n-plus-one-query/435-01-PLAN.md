   // BEFORE (N+1):
   for (const candidate of candidateSessions) {
     const linkedCount = deps.db.select({ cnt: drizzleCount() }).from(sessionWallets)
       .where(and(eq(sessionWallets.sessionId, candidate.id), ...)).get();
     if (linkedCount && linkedCount.cnt === targetWalletIds.length) { ... }
   }
   // AFTER (batch):
   const candidateIds = candidateSessions.map(c => c.id);

     ? deps.db

           cnt: drizzleCount(),

         ? Math.floor(candidate.expiresAt.getTime() / 1000)
         : (candidate.expiresAt as number);
